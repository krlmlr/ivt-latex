%% Titlepage definition
\newcommand{\createtitlepage}{
  \begin{titlepage}
   \begin{center}

    \noindent\textbf{\MakeUppercase{\mytitle}}

    \vspace{0.1in}

    \internauthortitlestring\\
    \textit{%
    \internaddresslist%
    }
    \end{center}

    \vspace{0.1in}
    \input{abstract}

    \vspace{0.1in}
    \textbf{Keywords: } \mykeywordsEN

    \vspace{0.1in}
    \textbf{Subject Area: } \mysubject

    \vspace{0.1in}
    \textbf{Preferred mode of presentation: } \myprefpresmode

    \vspace{0.1in}
    Request for the consideration of publication in the Proceedings: \textbf{\mypubinproc}
    \vfill
    \eject
  \end{titlepage}
}
